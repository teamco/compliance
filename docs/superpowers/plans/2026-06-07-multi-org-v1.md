# Admin-Owned Multi-Organization v1 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Allow one admin user to own and manage multiple organizations, with standards and gap analyses scoped per org.

**Architecture:** Drop the `unique(user_id)` constraint on `org_profiles`, add `gap_analyses` persistence table, and add `organization_members` scaffold table. New multi-org CRUD routes at `/notes/orgs/*` replace the single-org `/notes/org`. Gateway enforces ownership via CASL (`Organization` subject). Frontend gets an org switcher in the header (admin only), and all standards/gap queries are keyed by `activeOrgId` from a Zustand persist store.

**Tech Stack:** Supabase SQL migration, NestJS TCP microservices, `@casl/ability` MongoAbility, TanStack Query v5, Zustand persist, React 19 + shadcn

## Task 1: DB Migration

**Files:**
- Create: `supabase/migrations/20260607000007_multi_org.sql`

- [ ] **Step 1: Create migration file**

```sql
-- Drop unique constraint (one user can now own many orgs)
alter table public.org_profiles drop constraint if exists org_profiles_user_id_key;

-- organization_members — scaffold for v2 (no API/UI in v1)
create table if not exists public.organization_members (
  id        uuid primary key default gen_random_uuid(),
  org_id    uuid not null references public.org_profiles(id) on delete cascade,
  user_id   uuid not null references auth.users(id) on delete cascade,
  role      text not null default 'viewer' check (role in ('owner', 'admin', 'viewer')),
  joined_at timestamptz not null default now(),
  unique(org_id, user_id)
);
alter table public.organization_members enable row level security;
create policy "org_members_own"
  on public.organization_members for all
  using (auth.uid() = user_id);

-- gap_analyses — persisted gap analysis results
create table if not exists public.gap_analyses (
  id         uuid primary key default gen_random_uuid(),
  org_id     uuid not null references public.org_profiles(id) on delete cascade,
  user_id    uuid not null references auth.users(id) on delete cascade,
  doc_id     uuid references public.generated_standards(id) on delete set null,
  result     jsonb not null,
  risk_score int not null default 0,
  created_at timestamptz not null default now()
);
alter table public.gap_analyses enable row level security;
create policy "gap_analyses_own"
  on public.gap_analyses for all
  using (auth.uid() = user_id);

create index if not exists gap_analyses_org_id_idx on public.gap_analyses(org_id);
create index if not exists org_members_org_id_idx  on public.organization_members(org_id);
```

- [ ] **Step 1: Replace old org routes**

Remove `@Get('org')` and `@Put('org')` handlers entirely.

Add these new routes (after the frameworks routes):
```ts
import { BadRequestException, ForbiddenException, NotFoundException } from '@nestjs/common';
import { subject } from '@casl/ability';
import { AbilityFactory } from '../abilities/ability.factory';
import type { Organization, GapAnalysisResult } from '@icore/shared';

// In constructor, inject AbilityFactory:
constructor(
  private readonly notes: NotesClientService,
  private readonly ai: AiClientService,
  private readonly abilityFactory: AbilityFactory,
) {}

@Get('orgs')
@ApiOperation({ summary: 'List organizations owned by current user' })
async listOrgs(@Req() req: Request & { user?: VerifiedToken }) {
  const uid = this.uid(req);
  return this.notes.listOrganizations(uid);
}

@Post('orgs')
@ApiOperation({ summary: 'Create a new organization' })
@ApiBody({ schema: { type: 'object' } })
async createOrg(
  @Req() req: Request & { user?: VerifiedToken },
  @Body() body: OrganizationInput,
) {
  const uid = this.uid(req);
  return this.notes.createOrganization(uid, body);
}

@Get('orgs/:id')
@ApiOperation({ summary: 'Get organization by id' })
async getOrgById(
  @Req() req: Request & { user?: VerifiedToken },
  @Param('id') id: string,
) {
  const org = await this.notes.getOrganizationById(id);
  if (!org) throw new NotFoundException();
  this.checkOrgAccess(req, org, 'read');
  return org;
}

@Put('orgs/:id')
@ApiOperation({ summary: 'Update organization' })
@ApiBody({ schema: { type: 'object' } })
async updateOrg(
  @Req() req: Request & { user?: VerifiedToken },
  @Param('id') id: string,
  @Body() body: OrganizationInput,
) {
  const org = await this.notes.getOrganizationById(id);
  if (!org) throw new NotFoundException();
  this.checkOrgAccess(req, org, 'update');
  return this.notes.updateOrganization(id, body);
}

private checkOrgAccess(
  req: Request & { user?: VerifiedToken },
  org: Organization,
  action: 'read' | 'update' | 'delete',
): void {
  const ability = this.abilityFactory.forUser(req.user);
  if (!ability.can(action, subject('Organization', { id: org.id, userId: org.userId }))) {
    throw new ForbiddenException();
  }
}
```

```tsx
import { useEffect, useRef, useState } from 'react';
import { useNavigate } from '@tanstack/react-router';
import { Building2, ChevronDown, Plus } from 'lucide-react';
import { useTranslation } from 'react-i18next';
import { useIsAdmin } from '@icore/template-shared';
import { useOrganizations } from '@/queries/notes';
import { useActiveOrgStore } from '@/stores/active-org';
import { useOnClickOutside } from '@/hooks/useOnClickOutside';

export function OrgSwitcher() {
  const { t } = useTranslation();
  const isAdmin = useIsAdmin();
  const navigate = useNavigate();
  const { data: orgs } = useOrganizations();
  const { activeOrgId, setActiveOrgId } = useActiveOrgStore();
  const [open, setOpen] = useState(false);
  const ref = useRef<HTMLDivElement>(null);
  useOnClickOutside(ref, () => setOpen(false));

  // Auto-select first org on first load
  useEffect(() => {
    if (!activeOrgId && orgs && orgs.length > 0) {
      setActiveOrgId(orgs[0].id);
    }
  }, [orgs, activeOrgId, setActiveOrgId]);

  if (!isAdmin) return null;

  const activeOrg = orgs?.find((o) => o.id === activeOrgId);

  return (
    <div ref={ref} className="relative">
      <button
        type="button"
        onClick={() => setOpen((o) => !o)}
        className="flex items-center gap-2 px-3 h-8 rounded-md border border-border bg-surface text-sm text-foreground hover:bg-muted transition-colors cursor-pointer"
      >
        <Building2 size={14} className="text-muted-foreground shrink-0" />
        <span className="max-w-[140px] truncate">
          {activeOrg ? activeOrg.name : t('org.switchOrg')}
        </span>
        <ChevronDown size={12} className="text-muted-foreground shrink-0" />
      </button>

      {open && (
        <div className="absolute top-full mt-1 left-0 z-50 min-w-[180px] bg-surface border border-border rounded-lg shadow-lg py-1 text-sm">
          {(orgs ?? []).map((org) => (
            <button
              key={org.id}
              type="button"
              onClick={() => {
                setActiveOrgId(org.id);
                setOpen(false);
              }}
              className={[
                'w-full text-left px-3 py-1.5 truncate cursor-pointer transition-colors',
                org.id === activeOrgId
                  ? 'text-green-500 bg-green-500/5'
                  : 'text-foreground hover:bg-muted',
              ].join(' ')}
            >
              {org.name}
            </button>
          ))}

          {(orgs ?? []).length === 0 && (
            <p className="px-3 py-1.5 text-muted-foreground text-xs">{t('org.noOrgs')}</p>
          )}

          <div className="border-t border-border mt-1 pt-1">
            <button
              type="button"
              onClick={() => {
                setOpen(false);
                void navigate({ to: '/org', search: { create: 'true' } });
              }}
              className="w-full text-left flex items-center gap-2 px-3 py-1.5 text-muted-foreground hover:text-foreground hover:bg-muted cursor-pointer transition-colors"
            >
              <Plus size={12} />
              {t('org.createNew')}
            </button>
          </div>
        </div>
      )}
    </div>
  );
}
```
